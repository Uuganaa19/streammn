Supabase — үнэгүй, бодит database
supabase.com дээр бүртгэл үүсгэж шинэ project нэмнэ үү. Project Settings → API хэсгээс URL болон anon key авна уу.
Эсвэл "localStorage горим" сонгож нэн даруй ашиглаж болно.
✅ eyJ... (legacy) болон sb_publishable__... (шинэ) аль алийг оруулж болно
⚠️ Файлыг шууд нээсэн байна (file://)
Browser нь file:// горимд гадаад сүлжээний хүсэлт (Supabase API) хааж байна.
Шийдэл: PowerShell нээж Downloads хавтас дотор ажиллуулна уу:
npx serve .
Дараа нь http://localhost:3000/streamMN_admin.html хаягаар нээнэ үү.
Эсвэл доор "localStorage горим" сонгоно уу — Supabase-гүй ажиллана.

CREATE TABLE IF NOT EXISTS customers (
  id TEXT PRIMARY KEY DEFAULT gen_random_uuid()::text,
  name TEXT NOT NULL,
  email TEXT NOT NULL,
  phone TEXT,
  sub_type TEXT DEFAULT 'monthly',
  amount NUMERIC DEFAULT 12900,
  start_date DATE,
  end_date DATE,
  status TEXT DEFAULT 'pending',
  tx_code TEXT,
  invite_code TEXT,
  note TEXT,
  created_at TIMESTAMPTZ DEFAULT now(),
  updated_at TIMESTAMPTZ DEFAULT now()
);
CREATE TABLE IF NOT EXISTS payments (
  id TEXT PRIMARY KEY DEFAULT gen_random_uuid()::text,
  customer_id TEXT,
  customer_name TEXT,
  email TEXT,
  amount NUMERIC,
  sub_type TEXT,
  tx_code TEXT,
  bank TEXT DEFAULT 'M Bank',
  date DATE DEFAULT now(),
  status TEXT DEFAULT 'pending',
  bank_ref TEXT,
  verified_note TEXT,
  verified_at TIMESTAMPTZ,
  created_at TIMESTAMPTZ DEFAULT now()
);
-- Allow all access (admin only uses this)
ALTER TABLE customers ENABLE ROW LEVEL SECURITY;
ALTER TABLE payments  ENABLE ROW LEVEL SECURITY;
CREATE POLICY "allow_all" ON customers FOR ALL USING (true) WITH CHECK (true);
CREATE POLICY "allow_all" ON payments  FOR ALL USING (true) WITH CHECK (true);
